import java.time.LocalDate;
import java.time.Month;
import java.time.Period;
import java.time.temporal.ChronoUnit;

public class StanislavPopov {

    public static void main(String[] args) {

        // >>> Pers√∂nliche Basisdaten
        String fullName = "Stanislav Popov";
        String birthCity = "Luhansk";
        String birthCountry = "Ukraine";
        int birthYear = 2000;
        int birthDay = 26;
        Month birthMonth = Month.JULY;

        System.out.println(">>> [Analyse gestartet: Pers√∂nliche Basisdaten]\n");
        System.out.printf("-> Name: %s%n", fullName);
        System.out.printf("-> Geburtsort: %s, %s%n", birthCity, birthCountry);
        System.out.printf("-> Geburtsdatum: %d.%d.%d%n", birthDay, birthMonth.getValue(), birthYear);

        // >>> Realistisches Alter mit Period
        LocalDate today = LocalDate.now();
        LocalDate birthDate = LocalDate.of(birthYear, birthMonth, birthDay);
        Period lived = Period.between(birthDate, today);
        var livedYears = lived.getYears();
        var livedMonths = lived.getMonths();
        var livedDays = lived.getDays();

        System.out.println("\n>>> [Berechnung: Realistisch ‚Äì Bisher gelebte Zeit]");
        if (livedYears > 0 || livedMonths > 0 || livedDays > 0){
            System.out.printf("-> Gelebt (realistisch): %d Jahre, %d Monate, %d Tage%n", livedYears, livedMonths, livedDays);
        } else {
            System.out.println("-> Ung√ºltige Zeitspanne oder Zukunftsdatum");
        }

        // >>> ChronoUnit Details
        long totalWeeks = ChronoUnit.WEEKS.between(birthDate, today);
        long totalHours = ChronoUnit.HOURS.between(birthDate.atStartOfDay(), today.atStartOfDay());
        long totalMinutes = ChronoUnit.MINUTES.between(birthDate.atStartOfDay(), today.atStartOfDay());
        long totalSeconds = ChronoUnit.SECONDS.between(birthDate.atStartOfDay(), today.atStartOfDay());

        if (totalSeconds > 0) {
            System.out.printf("-> Das sind ungef√§hr: %d Wochen, %d Stunden, %d Minuten, %d Sekunden%n",
                totalWeeks, totalHours, totalMinutes, totalSeconds);
        } else {
            System.out.println("-> Keine chronologische Dauer ermittelt");
        }

        // >>> Alter (gesch√§tzt) via for
        var simulatedYears = today.getYear() - birthYear;
        if (today.getMonthValue() < birthMonth.getValue() ||
            (today.getMonthValue() == birthMonth.getValue() && today.getDayOfMonth() < birthDay)) {
            simulatedYears--;
        }

        var simulatedMonths = 0;
        for (int month = 1; month <= 12; month++) {
            simulatedMonths++;
            if (month == birthMonth.getValue()) break;
        }

        var simulatedDays = 0;
        for (int day = 1; day <= 31; day++) {
            simulatedDays++;
            if (day == birthDay) break;
        }

        System.out.println("\n>>> [Berechnung: Gesch√§tztes Alter]");
        if (simulatedYears >= 0 && simulatedMonths > 0 && simulatedDays > 0) {
            System.out.printf("-> Alter (gesch√§tzt): %d Jahre, %d Monate, %d Tage%n",
                simulatedYears, simulatedMonths, simulatedDays);
        } else {
            System.out.println("-> Fehler bei der gesch√§tzten Altersberechnung");
        }

        System.out.println(">>> Simulation abgeschlossen");

        // >>> Statusanalyse
        boolean migrated = true;
        boolean inRelationship = true;
        String partnerName = "Dora";
        String relationshipStatus = String.format("In Beziehung mit %s", partnerName);
        String currentCity = "Selb";

        System.out.println("\n>>> [Statusanalyse]");
        if (migrated) {
            System.out.println("[OK] Neuanfang in Deutschland");
        } else {
            System.out.println("[WARN] Noch im Herkunftsland");
        }

        if (inRelationship) {
            System.out.printf("[OK] Beziehungsstatus: %s%n", relationshipStatus);
        } else {
            System.out.println("[INFO] Aktuell nicht in einer Beziehung");
        }

        if (currentCity != null && !currentCity.isEmpty()) {
            System.out.printf("-> Aktueller Wohnort: %s%n", currentCity);
        } else {
            System.out.println("-> Wohnort nicht bekannt");
        }

        System.out.println("\n>>> Systemanalyse abgeschlossen: Pers√∂nliche Daten erfolgreich geladen");
    }
}
